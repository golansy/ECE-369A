	addi $t6, $zero, 0	# int direction = DIAG_UP;
	addi $t7, $zero, 0	# int curSAD = 0;
	
	LOOP:
        addi $t8, $zero, -1
		beq $t6, $t8, LOOP_END # if (direction == DONE) exit loop
		addiu $sp, $sp, -12
		sw $s5, 0($sp) # arg 5 is window
		sw $t1, 4($sp) # arg 6 is row
		sw $t2, 8($sp) # arg 7 is col
		addiu $sp, $sp, 12
		add $a0, $s3, $zero # a0 = window_width
		add $a1, $s2, $zero # a1 = window_height
		add $a2, $s1, $zero # a2 = frame_width
		add $a3, $s4, $zero # a3 = frame
		jal getSAD
		add $t7, $v0, $zero #curSAD = getSAD(window_width, window_height, frame_width, frame, window, row, col);
		
		bgt $t7, $t0, LOOP_PC # if (curSAD > minSAD) continue;
		add $t0, $t7, $zero # minSAD = curSAD;
		add $s6, $t1, $zero # minRow = row;
		add $s7, $t2, $zero # minCol - col;
		
		LOOP_PC:
		addiu $sp, $sp, -4
		sw $t6, 0($sp) # arg 5 is direction
		addiu $sp, $sp, 4
		add $a0, $t4, $zero # a0 = maxRow
		add $a1, $t3, $zero # a1 = maxCol
		add $a2, $t1, $zero # a2 = row
		add $a3, $t2, $zero # a3 = col
		jal getDirection
		add $t6, $v0, $zero # direction = getDirection(maxRow, maxCol, row, col, direction);
		
		add $a0, $t6, $zero # a0 = direction
		add $a1, $t1, $zero # a1 = row
		add $a2, $t2, $zero # a2 = col
		jal updatePosition
		add $t1, $v0, $zero # (row, col) = updatePosition(direction, row, col);
		add $t2, $v1, $zero
		j LOOP
	LOOP_END:
	
	add $v0, $s6, $zero # return (minRow, minCol)
	add $v1, $s7, $zero

	#Standard Epilogue
	lw $ra, 4($sp)
	lw $fp, 0($sp)
	addiu $sp, $sp, 24
	jr $ra
    
.text
.globl updatePosition

updatePosition:
	#Standard Prologue
	addiu $sp, $sp, -24
	sw $fp, 0($sp)
	sw $ra, 4($sp)
	addiu $fp, $sp, 20
	
	add $v0, $a1, $zero
	add $v1, $a2, $zero
	
	beq $a0, $zero, UPDATE_DIAG_UP	# if direction == DIAG_UP
    	addi $t8, $zero, 1
	beq $a0, $t8, UPDATE_RIGHT		# if direction == RIGHT
    	addi $t8, $zero, 2
	beq $a0, $t8, UPDATE_DIAG_LEFT	# if direction == DIAG_LEFT
    	addi $t8, $zero, 3
	beq $a0, $t8, UPDATE_DOWN		# if direction == DOWN
	
	UPDATE_DIAG_UP:
	addi $v0, $a1, -1	# row = row - 1;
	UPDATE_RIGHT:
	addi $v1, $a2, 1	# col = col + 1;
	j UPDATE_EXIT		# return (row, col);
	
	UPDATE_DIAG_LEFT:
	addi $v1, $a2, -1	# col = col - 1;
	UPDATE_DOWN:
	addi $v0, $a1, 1	# row = row + 1;
				# return (row, col);
	UPDATE_EXIT:
	#Standard Epilogue
	lw $ra, 4($sp)
	lw $fp, 0($sp)
	addiu $sp, $sp, 24
	jr $ra
	
.text
.globl getDirection

getDirection:
	#Standard Prologue
	addiu $sp, $sp, -28
	sw $fp, 0($sp)
	sw $ra, 4($sp)
	addiu $fp, $sp, 24
	
	addiu $sp, $sp, -32 #adds 8 words to the stack frame to save the 8 $sX registers
	sw $s7, 0($sp) #saves $s7
	sw $s6, 4($sp) #saves $s6
	sw $s5, 8($sp) #saves $s5
	sw $s4, 12($sp) #saves $s4
	sw $s3, 16($sp) #saves $s3
	sw $s2, 20($sp) #saves $s2
	sw $s1, 24($sp) #saves $s1
	sw $s0, 28($sp) #saves $s0
	
	
	lw $s0, 0($fp) # loads in the fifth argument, direction
	
	sub $s1, $a0, $a2 # s1 = maxRow - row;
	sub $s2, $a1, $a3 # s2 = maxCol - col;
	add $s3, $s1, $s2 # s3 = (maxRow - row) + (maxCol - col);
	beq $s3, $zero, DIR_RET_DONE # if (maxRow == row && maxCol == col) return DONE;
	
	beq $s0, $zero, DIR_DIAG_UP # if (direction == DIAG_UP)
    	addi $t8, $zero, 1
	beq $s0, $t8, DIR_RIGHT # if (direction == RIGHT)
    	addi $t8, $zero, 2
	beq $s0, $t8, DIR_DIAG_LEFT # if (direction == DIAG_LEFT)
	addi $t8, $zero, 3
	beq $s0, $t8, DIR_DOWN # if (direction == DOWN)
	
	DIR_DIAG_UP:
		beq $s2, $zero, DIR_RET_DOWN # if (col == maxCol) return DOWN;
		beq $a2, $zero, DIR_RET_RIGHT # if (row == 0) return RIGHT;
		j DIR_RET_DIAG_UP	# else return DIAG_UP;
	
	DIR_RIGHT:
		beq $a0, $zero, DIR_RET_RIGHT # if (maxRow == 0) return RIGHT;
		beq $a2, $zero, DIR_RET_DIAG_LEFT # if (row == 0) return DIAG_LEFT;
		j DIR_RET_DIAG_UP	# else return DIAG_UP
	
	DIR_DIAG_LEFT:
		beq $s1, $zero, DIR_RET_RIGHT # if (row == maxRow) return RIGHT;
		beq $a3, $zero, DIR_RET_DOWN # if (col == 0) return DOWN;
		j DIR_RET_DIAG_LEFT	# else return DIAG_LEFT;
	
	DIR_DOWN:
		beq $a1, $zero, DIR_RET_DOWN # if (maxCol == 0) return DOWN;
		beq $a3, $zero, DIR_RET_DIAG_UP # if (col == 0) return DIAG_UP;
		j DIR_RET_DIAG_LEFT # else return DIAG_LEFT;
		
	DIR_RET_DONE:
		addi $v0, $zero, -1 # return DONE;
		j DIR_EXIT
	DIR_RET_DIAG_UP:
		addi $v0, $zero, 0 # return DIAG_UP;
		j DIR_EXIT
	DIR_RET_RIGHT:
		addi $v0, $zero, 1 # return RIGHT;
		j DIR_EXIT
	DIR_RET_DIAG_LEFT:
		addi $v0, $zero, 2 # return DIAG_LEFT;
		j DIR_EXIT
	DIR_RET_DOWN:
		addi $v0, $zero, 3 # return DOWN;
		j DIR_EXIT
	
	DIR_EXIT:
	lw $s0, 28($sp) #Restores $s0
	lw $s1, 24($sp) #Restores $s1
	lw $s2, 20($sp) #Restores $s2
	lw $s3, 16($sp) #Restores $s3
	lw $s4, 12($sp) #Restores $s4
	lw $s5, 8($sp) #Restores $s5
	lw $s6, 4($sp) #Restores $s6
	lw $s7, 0($sp) #Restores $s7
	addiu $sp, $sp, 32 #Reduces the stack frame size by 8 words
	
	#Standard Epilogue
	lw $ra, 4($sp)
	lw $fp, 0($sp)
	addiu $sp, $sp, 28
	jr $ra
	
.text
.globl getSAD

getSAD:
	#Standard Prologue
	addiu $sp, $sp, -36
	sw $fp, 0($sp)
	sw $ra, 4($sp)
	addiu $fp, $sp, 24
	
	addiu $sp, $sp, -32 #adds 8 words to the stack frame to save the 8 $sX registers
	sw $s7, 0($sp) #saves $s7
	sw $s6, 4($sp) #saves $s6
	sw $s5, 8($sp) #saves $s5
	sw $s4, 12($sp) #saves $s4
	sw $s3, 16($sp) #saves $s3
	sw $s2, 20($sp) #saves $s2
	sw $s1, 24($sp) #saves $s1
	sw $s0, 28($sp) #saves $s0
	
	lw $s0, 0($fp) # loads in 5th argument, window
	lw $s1, 4($fp) # loads in 6th argument, curRow
	lw $s2, 8($fp) # loads in 7th argument, curCol
	
	addi $s3, $zero, 0 	#int curFrameNum = 0;
	addi $s4, $zero, 0	#int curWindowNum = 0;
	addi $s5, $zero, 0	#int curSAD = 0;
	addi $s6, $zero, 0	# int row = 0;
	
	SAD_ROW_LOOP:
		#bge $s6, $a1, SAD_ROW_LOOP_END # if (row >= window_height) exit loop
		sub $t8, $s6, $a1
		bgez $t8, SAD_ROW_LOOP_END
		addi $s7, $zero, 0	# int col = 0;
		SAD_COL_LOOP:
			#bge $s7, $a0, SAD_COL_LOOP_END # if (col >= window_width) exit loop
			sub $t8, $s7, $a0
			bgez $t8, SAD_COL_LOOP_END
			
			add $t9, $s6, $s1	# index = row + curRow;
			mul $t9, $t9, $a2	# index = index * frame_width;
			add $t9, $t9, $s7	# index = index + col;
			add $t9, $t9, $s2	# index = index + curCol;
			
			addi $t8, $zero, 4
			mul $t9, $t9, $t8   #multiplying $t9 by 4

			add $t9, $t9, $a3
			lw $s3, 0($t9)	# curFrameNum = frame[(row + curRow) * frame_width + col + curCol];
			
			mul $t9, $s6, $a0	# index = row * window_width;
			add $t9, $t9, $s7	# index = index + col;
			
			addi $t8, $zero, 4
			mul $t9, $t9, $t8   #multiplying $t9 by 4
			
			add $t9, $t9, $s0
			lw $s4, 0($t9)	# curWindowNum = window[row * window_width + col];
			
			#bge $s3, $s4, FRAME_GE # if (curFrameNum > curWindowNum) goto FRAME_GE else continue
			sub $t8, $s3, $s4
			bgtz $t8, FRAME_GE
			add $s5, $s5, $s4 # curSAD = curSAD + curWindowNum;
			sub $s5, $s5, $s3 # curSAD = curSAD - curFrameNum;
			j SAD_COL_LOOP_PC
			
			FRAME_GE:
				add $s5, $s5, $s3 # curSAD = curSAD + curFrameNum;
				sub $s5, $s5, $s4 # curSAD = curSAD - curWindowNum;
			
			SAD_COL_LOOP_PC:
			addi $s7, $s7, 1	# col++;
			j SAD_COL_LOOP
		SAD_COL_LOOP_END:
		addi $s6, $s6, 1	# row++;
		j SAD_ROW_LOOP
	SAD_ROW_LOOP_END:
	
	add $v0, $s5, $zero # return curSAD;
	
	lw $s0, 28($sp) #Restores $s0
	lw $s1, 24($sp) #Restores $s1
	lw $s2, 20($sp) #Restores $s2
	lw $s3, 16($sp) #Restores $s3
	lw $s4, 12($sp) #Restores $s4
	lw $s5, 8($sp) #Restores $s5
	lw $s6, 4($sp) #Restores $s6
	lw $s7, 0($sp) #Restores $s7
	addiu $sp, $sp, 32 #Reduces the stack frame size by 8 words
	
	#Standard Epilogue
	lw $ra, 4($sp)
	lw $fp, 0($sp)
	addiu $sp, $sp, 36
	jr $ra
